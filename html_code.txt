        // Cart
        function addToCart(id) {

                alert('කාට් එකට එකතු විය!');
            } else if (cart.find(c => c.id === id)) {

            const cartItemsDiv = document.getElementById('cartItems');
            if (cart.length === 0) {
                cartItemsDiv.innerHTML = '<p style="text-align:center; color:#7f8c8d;">කාට් එක හිස්ය</p>';

        function processPayment() {
            if (!selectedPaymentMethod) {
                alert('කරුණාකර ගෙවීමේ ක්‍රමයක් තෝරන්න!');
                return;
            }
            
            const order = {
                id: Date.now(),
                item: currentPaymentItem,
                paymentMethod: selectedPaymentMethod,
                date: new Date().toISOString().split('T')[0],
                status: 'සම්පූර්ණයි'
            };
            
            orders.unshift(order);
            saveOrders();
            
            // Remove from cart if exists
            cart = cart.filter(c => c.id !== currentPaymentItem.id);
            saveCart();
            
            alert('✅ ගෙවීම සාර්ථකයි! ඔබගේ ඇණවුම සම්පූර්ණ විය.');
            closeModal('paymentModal');
        }

        }
